import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import os

# Set style for beautiful plots
sns.set_style("whitegrid")
plt.style.use('ggplot')

# Load the Excel file
file_path = 'Desktop/Warehouse_Inventory_Tracker.xlsx'
df_raw = pd.read_excel(file_path, sheet_name='Inventory Log')

# Clean and reformat the DataFrame
df = df_raw.iloc[1:].copy()
df.columns = df_raw.iloc[0]
df = df.dropna(how='all')
df = df.rename(columns={
    df.columns[0]: "Warehouse",
    "Floor": "Floor",
    "Room": "Room",
    "Section": "Section",
    "Item Name": "Item Name",
    "Quantity": "Quantity",
    "Unit": "Unit",
    "Status": "Status"
})
df["Quantity"] = pd.to_numeric(df["Quantity"], errors='coerce')
df.head()

# Create output directory if it doesn't exist
output_dir = os.path.expanduser('~/Desktop/Warehouse_Plots')
os.makedirs(output_dir, exist_ok=True)

# Custom color palettes
floor_palette = sns.color_palette("Blues_d", len(df['Floor'].unique()))
status_palette = sns.color_palette("pastel")
top_items_palette = sns.color_palette("viridis", 10)
status_count_palette = sns.color_palette("rocket_r", len(df['Status'].unique()))

# Figure 1: Inventory Overview
fig1, (ax1, ax2) = plt.subplots(1, 2, figsize=(18, 7), gridspec_kw={'wspace': 0.3})

# Subplot 1: Total quantity by floor/location
floor_totals = df.groupby('Floor')['Quantity'].sum().sort_values(ascending=False)
top_floor = floor_totals.idxmax()
top_value = floor_totals.max()

sns.barplot(x=floor_totals.index, y=floor_totals.values, ax=ax1, palette=floor_palette)
ax1.set_title('Total Inventory Quantity by Location\n', fontsize=14, fontweight='bold')
ax1.set_xlabel('Location (Floor)', fontsize=12)
ax1.set_ylabel('Total Quantity', fontsize=12)
ax1.tick_params(axis='x', rotation=45)
ax1.grid(True, linestyle='--', alpha=0.6)

# Add value labels on top of bars
for p in ax1.patches:
    ax1.annotate(f"{p.get_height():,.0f}", 
                (p.get_x() + p.get_width() / 2., p.get_height()), 
                ha='center', va='center', 
                xytext=(0, 5), 
                textcoords='offset points',
                fontsize=9)

# Subplot 2: Item status distribution
status_counts = df['Status'].value_counts()
dominant_status = status_counts.idxmax()

wedges, texts, autotexts = ax2.pie(status_counts, 
                                  autopct='%1.1f%%', 
                                  startangle=90, 
                                  colors=status_palette,
                                  wedgeprops={'linewidth': 1, 'edgecolor': 'white'},
                                  textprops={'fontsize': 10})

ax2.set_title('Inventory Condition Distribution\n', fontsize=14, fontweight='bold')
ax2.legend(wedges, status_counts.index,
          title="Status",
          loc="center left",
          bbox_to_anchor=(1, 0, 0.5, 1))

# Figure 2: Item Analysis
fig2, (ax3, ax4) = plt.subplots(1, 2, figsize=(18, 7), gridspec_kw={'wspace': 0.4})

# Subplot 3: Top 10 Items by Quantity
top_items = df.groupby('Item Name')['Quantity'].sum().sort_values(ascending=False).head(10)
top_item_name = top_items.index[0]
top_item_qty = top_items.values[0]

sns.barplot(x=top_items.values, y=top_items.index, ax=ax3, palette=top_items_palette)
ax3.set_title('Top 10 Items by Inventory Quantity\n', fontsize=14, fontweight='bold')
ax3.set_xlabel('Total Quantity', fontsize=12)
ax3.set_ylabel('Item Name', fontsize=12)
ax3.grid(True, linestyle='--', alpha=0.6)

# Add value labels on bars
for p in ax3.patches:
    width = p.get_width()
    ax3.annotate(f"{width:,.0f}",
                (width, p.get_y() + p.get_height()/2.),
                ha='left', va='center',
                xytext=(5, 0),
                textcoords='offset points',
                fontsize=9)

# Subplot 4: Status count
status_order = df['Status'].value_counts().index
sns.countplot(data=df, y='Status', order=status_order, ax=ax4, palette=status_count_palette)
ax4.set_title('Inventory Condition Count\n', fontsize=14, fontweight='bold')
ax4.set_xlabel('Number of Items', fontsize=12)
ax4.set_ylabel('Condition', fontsize=12)
ax4.grid(True, linestyle='--', alpha=0.6)

# Add value labels on bars
for p in ax4.patches:
    width = p.get_width()
    ax4.annotate(f"{width:,.0f}",
                (width, p.get_y() + p.get_height()/2.),
                ha='left', va='center',
                xytext=(5, 0),
                textcoords='offset points',
                fontsize=9)

# Save figures with high quality
fig1.savefig(os.path.join(output_dir, 'Figure1_Inventory_Overview.png'), 
            dpi=300, bbox_inches='tight', facecolor='white')
fig2.savefig(os.path.join(output_dir, 'Figure2_Item_Analysis.png'), 
            dpi=300, bbox_inches='tight', facecolor='white')

plt.show()
